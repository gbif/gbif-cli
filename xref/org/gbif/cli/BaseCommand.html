35   * This base command can be used for all Commands that want to have their parameters validated and who want to allow
36   * specifying configuration data in a JSON or YAML file.

87    @Override
88    public Optional<String> getUsage() {
89      JCommander jCommander = new JCommander();
90      jCommander.addConverterFactory(new ConverterFactory());
91      jCommander.setProgramName(name);
92      jCommander.addObject(new GenericParameters());
93      jCommander.addObject(getParameterObject());
94  
95      StringBuilder sb = new StringBuilder();
96      jCommander.usage(sb);
97  
98      String sep = System.getProperty("line.separator");
99      try {
100       JsonSchema jsonSchema = MAPPER.generateJsonSchema(getConfigurationObject().getClass());
101       sb.append(sep);
102       sb.append("The configuration file has the following JSON schema but accepts either YAML or JSON:");
103       sb.append(sep);
104       sb.append(jsonSchema);
105     } catch (Exception e) {
106       LOG.debug("Error generating JSON schema, this is most likely a programming error", e);
107     }
108 
109     return Optional.of(sb.toString());
110   }

112   @SuppressWarnings("UseOfSystemOutOrSystemErr")
113   @Override
114   public final void run(String... arguments) {
115     // First we process the generic parameters which might cause us to bail out early if there is an error or if the
116     // user requested us to print usage instructions
117     GenericParameters genericParameters = new GenericParameters();
118     JCommander jCommander = new JCommander(genericParameters);
119     jCommander.addConverterFactory(new ConverterFactory());
120     jCommander.setAcceptUnknownOptions(true);
121     processCommandLineParameters(jCommander, arguments);
122     if (genericParameters.help) {
123       System.out.println(getUsage().get());
124       return;
125     }
126 
127     processConfigurationFiles(genericParameters.configurationFiles);
128 
129     List<String> remainingArgs = jCommander.getUnknownOptions();
130     jCommander = new JCommander(getParameterObject());
131     jCommander.addConverterFactory(new ConverterFactory());
132     processCommandLineParameters(jCommander, remainingArgs.toArray(new String[remainingArgs.size()]));
133 
134     configureLogging(genericParameters);
135     validateObjects();
136 
137     doRun();
138   }

150   /**
151    * We take every passed in configuration file, test if it exists and try to write it to the Object provided by {@link
152    * #getConfigurationObject()}. This method prints errors it encounters.
153    *
154    * @param configurationFiles to process
155    */
156   private void processConfigurationFiles(Iterable<String> configurationFiles) {
157     for (String fileName : configurationFiles) {
158       File file = new File(fileName);
159       if (!file.exists()) {
160         System.err.println("Error reading configuration file [" + fileName + "] because it does not exist");
161         throw new CommandException("Error reading configuration file [" + fileName + "] because it does not exist");
162       }
163       ObjectReader reader = MAPPER.readerForUpdating(getConfigurationObject());
164       try {
165         reader.readValue(file);
166       } catch (IOException e) {
167         System.err.println("Error reading configuration file [" + fileName + "]");
168         throw new CommandException("Error reading configuration file [" + fileName + "]", e);
169       } catch (ParserException e) {
170         System.err.println("Error reading configuration file [" + fileName + "]");
171         throw new CommandException("Error reading configuration file [" + fileName + "]", e);
172       }
173     }
174   }

176   /**
177    * Reconfigures Logback.
178    * <p/>
179    * Usually logback loads a <em>logback.xml</em> file but we change that. We allow a command line parameter to
180    * override the config file to use. If that isn't specified or doesn't exist we try to load a file called
181    * <em>logback-$commandName.xml</em> and if that too doesn't exist we are using the default.
182    * <p/>
183    * Additionally we set the ROOT log level to the one that was provided, if any, or to DEBUG if requested otherwise we
184    * leave it at the default.
185    */
186   private void configureLogging(GenericParameters parameters) {
187     // Set up JUL logging to go through SLF4J
188     SLF4JBridgeHandler.removeHandlersForRootLogger();
189     SLF4JBridgeHandler.install();
190 
191     URL loggingConfigUrl = null;
192 
193     // Try to load a logback configuration file from the file system if given on the command line
194     if (parameters.logbackConfig != null && !parameters.logbackConfig.isEmpty()) {
195       File configFile = new File(parameters.logbackConfig);
196       if (!configFile.exists() || !configFile.isFile()) {
197         throw new CommandException("Logback configuration file does not exist");
198       }
199       try {
200         loggingConfigUrl = configFile.toURI().toURL();
201       } catch (MalformedURLException e) {
202         // This should not happen because we already checked for existence
203         LOG.debug("Failed to create URL for logback configuration file");
204         throw new CommandException(e);
205       }
206     }
207 
208     // Try to automatically load a logback configuration file named after the command
209     if (loggingConfigUrl == null) {
210       loggingConfigUrl = Loader.getResource("logback-" + getName() + ".xml", Loader.getClassLoaderOfObject(this));
211     }
212 
213     // If we got a configuration file reconfigure logback now
214     if (loggingConfigUrl != null) {
215       LoggerContext context = (LoggerContext) LoggerFactory.getILoggerFactory();
216       context.reset();
217       JoranConfigurator joran = new JoranConfigurator();
218       joran.setContext(context);
219       try {
220         joran.doConfigure(loggingConfigUrl);
221       } catch (JoranException e) {
222         StatusPrinter.printInCaseOfErrorsOrWarnings(context);
223       }
224     }
225 
226     // Change log levels if needed
227     if (parameters.verbose || (parameters.logLevel != null && !parameters.logLevel.isEmpty())) {
228       ch.qos.logback.classic.Logger root =
229         (ch.qos.logback.classic.Logger) LoggerFactory.getLogger(Logger.ROOT_LOGGER_NAME);
230       root.setLevel(Level.toLevel(parameters.logLevel));
231 
232       if (parameters.verbose) {
233         root.setLevel(Level.DEBUG);
234       }
235     }
236   }

238   /**
239    * This validates the parameter and configuration object and prints violations if there are some. If the objects are
240    * exactly the same it will only be validated once.
241    */
242   @SuppressWarnings("ObjectEquality")
243   private void validateObjects() {
244     validateObject(VALIDATOR, getConfigurationObject());
245     if (getParameterObject() != getConfigurationObject()) {
246       validateObject(VALIDATOR, getParameterObject());
247     }
248   }

250   /**
251    * Validates a single object, printing any violations it encounters.
252    *
253    * @param validator to use
254    * @param object    to validate
255    */
256   private void validateObject(Validator validator, Object object) {
257     Set<ConstraintViolation<Object>> violations = validator.validate(object);
258     if (!violations.isEmpty()) {
259       System.err.println("Failed validation of command line parameters:");
260       for (ConstraintViolation<Object> violation : violations) {
261         System.err.println("  " + violation.getPropertyPath() + ": " + violation.getMessage());
262       }
263       throw new CommandException("Failed validation");
264     }
265   }

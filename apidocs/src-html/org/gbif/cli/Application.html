001package org.gbif.cli;
002
003import java.io.OutputStream;
004import java.io.PrintWriter;
005import java.net.URL;
006import java.util.Arrays;
007import java.util.Map;
008import java.util.ServiceLoader;
009
010import ch.qos.logback.classic.LoggerContext;
011import ch.qos.logback.classic.util.ContextInitializer;
012import com.google.common.base.Optional;
013import com.google.common.collect.Maps;
014import com.google.common.io.Closeables;
015import org.slf4j.Logger;
016import org.slf4j.LoggerFactory;
017
018import static com.google.common.base.Preconditions.checkNotNull;
019
020/**
021 * This is the main entry point into any CLI application.
022 */
023public class Application {
024
025  private static final Logger LOG = LoggerFactory.getLogger(Application.class);
026
027  // The supported commands in the order they were added
028  private final Map<String, Command> commands = Maps.newTreeMap();
029
030  protected Application() {
031
032  }
033
034  /**
035   * Adds a command to this Application. If another command with the same name was already registered it will be
036   * replaced by this new one.
037   *
038   * @param command to add or replace
039   */
040  protected final void addCommand(Command command) {
041    checkNotNull(command);
042    commands.put(command.getName(), command);
043    LOG.debug("Added Command [{}]", command.getName());
044  }
045
046  /**
047   * Can be overridden by subclasses to initialize themselves for example by adding commands.
048   */
049  public void initialize() {}
050
051  /**
052   * Main entry point. This method searches for other commands on the classpath using the {@link ServiceLoader} and
053   * parses the command line to see which command to run.
054   *
055   * @param arguments command line arguments
056   *
057   * @return exit code
058   */
059  @SuppressWarnings("UseOfSystemOutOrSystemErr")
060  public final int run(String... arguments) {
061    configureLogback();
062
063    // Try to load additional commands
064    ServiceLoader<Command> loader = ServiceLoader.load(Command.class);
065    for (Command command : loader) {
066      addCommand(command);
067    }
068
069    initialize();
070
071    if (arguments.length == 0) {
072      printUsage(System.err);
073      return 1;
074    }
075
076    // Command has to be the very first argument
077    String commandName = arguments[0];
078    if (!commands.containsKey(commandName)) {
079      System.err.println("The command name you supplied is not a valid command.");
080      System.err.println("You supplied: [" + commandName + "]");
081      if (commands.isEmpty()) {
082        System.err.println("There are no valid commands. This is most likely a programming error.");
083        System.err.println("Hint: Maybe you forgot to annotate your commands with '@MetaInfServices(Command.class)'?");
084      } else {
085        System.err.println("The valid commands are:");
086      }
087      for (String availableCommand : commands.keySet()) {
088        System.err.println(" - " + availableCommand);
089      }
090      return 1;
091    }
092
093    // Copy the remaining arguments (minus command name)
094    String[] commandArguments = Arrays.copyOfRange(arguments, 1, arguments.length);
095
096    Command command = commands.get(commandName);
097    try {
098      command.run(commandArguments);
099    } catch (Exception t) {
100      System.err.println("Command threw exception");
101      t.printStackTrace();
102      return 1;
103    }
104    return 0;
105  }
106
107  /**
108   * This checks if the user has provided a logback configuration file using the usual logback loading mechanism. If he
109   * does not logback will already have run {@link ch.qos.logback.classic.BasicConfigurator#configureDefaultContext();}
110   * which means we're logging to the console at DEBUG level. To prevent that we try the same mechanism that logback
111   * uses to find a configuration file. If it can't find this file then it resets logback so that it doesn't print
112   * anything.
113   * <p/>
114   * The drawback here is that it doesn't take into account the configurability of logback files using the BaseCommand.
115   */
116  private void configureLogback() {
117    LoggerContext context = (LoggerContext) LoggerFactory.getILoggerFactory();
118    ContextInitializer ci = new ContextInitializer(context);
119    URL loggingConfigUrl = ci.findURLOfDefaultConfigurationFile(false);
120
121    if (loggingConfigUrl == null) {
122      context.reset();
123    }
124  }
125
126  private void printUsage(OutputStream os) {
127    PrintWriter pw = new PrintWriter(os);
128    try {
129      pw.println("Usage");
130
131      for (Map.Entry<String, Command> commandEntry : commands.entrySet()) {
132        pw.println();
133        pw.append("Command: ").println(commandEntry.getKey());
134        pw.println("---------------------------------------");
135
136        Optional<String> usage = commandEntry.getValue().getUsage();
137        pw.println(usage.or("This command does not have any options"));
138      }
139    } catch (Exception e) {
140      System.err.println("Exception during generation of usage instructions");
141      e.printStackTrace();
142    } finally {
143      pw.flush();
144      Closeables.closeQuietly(pw);
145    }
146  }
147
148  public static void main(String[] args) {
149    System.exit(new Application().run(args));
150  }
151
152}

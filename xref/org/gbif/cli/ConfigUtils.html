1   package org.gbif.cli;

3   import java.lang.reflect.Field;
4   import java.lang.reflect.Modifier;
5   import java.util.Collection;
6   import java.util.Properties;
7   
8   import com.google.common.base.Strings;
9   
10  /**
11   * Utils class to extract properties from a configuration bean instance.
12   * By default this class adds all public field names to the generated properties recursively following nesting beans.
13   * Two field annotations are supported to alter this behavior:
14   * {@link IgnoreProperty} tells this util to ignore the field from the resulting properties and
15   * {@link PropertyName} can be used to assign a different local name than the default field name.
16   */
17  public class ConfigUtils {
18  
19    /**
20     * Builds properties from a configuration bean using no namespace prefix.
21     *
22     * @param config the configuration object to inspect
23     *
24     * @return properties instance derived
25     */
26    public static Properties toProperties(Object config) {
27      return toProperties("", config);
28    }
29  
30    /**
31     * Builds properties from a configuration bean using a given namespace prefix which is added to all properties.
32     *
33     * @param config the configuration object to inspect
34     *
35     * @return properties instance derived
36     */
37    public static Properties toProperties(String prefix, Object config) {
38      Properties props = new Properties();
39  
40      Field[] fields = config.getClass().getDeclaredFields();
41      for (Field f : fields) {
42        // ignore private fields
43        if (Modifier.isPublic(f.getModifiers()) && f.getAnnotation(IgnoreProperty.class) == null) {
44          try {
45            final String localName =
46              f.getAnnotation(PropertyName.class) == null ? f.getName() : f.getAnnotation(PropertyName.class).value();
47            final String qualName = Strings.isNullOrEmpty(prefix) ? localName : prefix + "." + localName;
48            Object val = f.get(config);
49            if (val == null) {
50              val = "";
51            }
52            // a gbif class? then we need to recursively inspect deeper
53            if (f.getType().getCanonicalName().startsWith("org")) {
54              Properties subProps = toProperties(qualName, val);
55              props.putAll(subProps);
56            } else if (!Collection.class.isAssignableFrom(f.getType())) {
57              props.setProperty(qualName, val.toString());
58            }
59          } catch (IllegalAccessException ignored) {
60            // should never get here cause its accessible
61          }
62        }
63      }
64  
65      return props;
66    }
67  
68    private ConfigUtils() {
69      throw new UnsupportedOperationException("Can't initialize class");
70    }
71  
72  }

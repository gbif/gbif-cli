001package org.gbif.cli.service;
002
003import org.gbif.cli.BaseCommand;
004
005import com.google.common.util.concurrent.ListenableFuture;
006import com.google.common.util.concurrent.Service;
007import com.google.common.util.concurrent.UncheckedExecutionException;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011/**
012 * This provides a convenient way to manage long-running services from a CLI. It is based on the {@link BaseCommand}
013 * and thus inherits all its possibilities to configure a command.
014 * <p/>
015 * You mostly need to implement the {@link #getService()} method to return a Guava {@link Service} class. This will
016 * then be started and stopped on JVM shutdown.
017 */
018public abstract class ServiceCommand extends BaseCommand {
019
020  private static final Logger LOG = LoggerFactory.getLogger(ServiceCommand.class);
021
022  private final Thread mainThread = Thread.currentThread();
023
024  private volatile boolean running = true;
025
026  protected ServiceCommand(String name) {
027    super(name);
028  }
029
030  /**
031   * This method will be called after the command line arguments and the configuration file have been processed.
032   * <p/>
033   * The service will be started and on shutdown of the JVM it will also try to execute an orderly shutdown of the
034   * service (using {@link Service#start()} and {@link Service#stop()} respectively).
035   */
036  protected abstract Service getService();
037
038  @Override
039  protected final void doRun() {
040    final ForwardingJoinableService service = new ForwardingJoinableService(getService());
041
042    Runtime.getRuntime().addShutdownHook(new ShutdownThread(service));
043
044    LOG.info("Starting service...");
045    try {
046      service.startAndWait();
047      LOG.info("Service started, now in state [{}]", service.state());
048    } catch (UncheckedExecutionException e) {
049      LOG.warn("Service failed to start", e);
050    }
051
052    try {
053      while (service.state() == Service.State.RUNNING) {
054        service.join();
055        LOG.debug("Got notified by service, service terminated");
056      }
057    } catch (InterruptedException e) {
058      LOG.debug("We were interrupted waiting for the service to stop, will rejoin", e);
059    }
060
061    synchronized (mainThread) {
062      running = false;
063      mainThread.notifyAll();
064    }
065
066    LOG.debug("Application shutting down");
067  }
068
069  /**
070   * This thread is intended to be passed to {@link Runtime#addShutdownHook(Thread)} and it tries to stop the service
071   * that it's being passed.
072   */
073  @SuppressWarnings("ClassExplicitlyExtendsThread")
074  private class ShutdownThread extends Thread {
075
076    private final ForwardingJoinableService service;
077
078    private ShutdownThread(ForwardingJoinableService service) {
079      this.service = service;
080    }
081
082    @Override
083    public void run() {
084      LOG.debug("Shutdown Hook called");
085      try {
086        if (service.isRunning()) {
087          LOG.info("Service stopping...");
088          ListenableFuture<Service.State> stopFuture = service.stop();
089          // TODO: Maybe wait with a timeout here after which we force close?
090          stopFuture.get();
091          LOG.info("Service stopped");
092        }
093        // TODO: Maybe wait with a timeout here after which we force close?
094        LOG.debug("Waiting for main Thread to exit");
095        synchronized (mainThread) {
096          while (running) {
097            mainThread.wait();
098          }
099        }
100        LOG.debug("Main thread exited");
101      } catch (Exception e) {
102        LOG.debug("Caught exception in shutdown hook", e);
103      }
104    }
105  }
106}

001package org.gbif.cli;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import com.beust.jcommander.Parameter;
007import com.google.common.base.Objects;
008
009public class GenericParameters {
010
011  @Parameter(
012    names = {"-h", "--help"},
013    description = "Prints help about this command",
014    help = true)
015  public boolean help;
016
017  @Parameter(
018    names = {"-c", "--conf"},
019    description = "Configuration files to use. Later ones override earlier ones",
020    variableArity = true)
021  public List<String> configurationFiles = new ArrayList<String>();
022
023  @Parameter(
024    names = "--log-config",
025    description = "Name of a logback XML configuration file to use")
026  public String logbackConfig;
027
028  @Parameter(
029    names = "--log-level",
030    description = "Name of the log level to use for the root logger, if an invalid name is given DEBUG will be used")
031  public String logLevel;
032
033  @Parameter(
034    names = {"-v", "--verbose"},
035    description = "Enables DEBUG logging. This overrides any --log-level setting")
036  public boolean verbose;
037
038  @Override
039  public String toString() {
040    return Objects.toStringHelper(this)
041      .add("help", help)
042      .add("configurationFiles", configurationFiles)
043      .add("logbackConfig", logbackConfig)
044      .add("loglevel", logLevel)
045      .add("verbose", verbose)
046      .toString();
047  }
048
049}

001package org.gbif.cli;
002
003import java.lang.reflect.Field;
004import java.lang.reflect.Modifier;
005import java.util.Collection;
006import java.util.Properties;
007
008import com.google.common.base.Strings;
009
010/**
011 * Utils class to extract properties from a configuration bean instance.
012 * By default this class adds all public field names to the generated properties recursively following nesting beans.
013 * Two field annotations are supported to alter this behavior:
014 * {@link IgnoreProperty} tells this util to ignore the field from the resulting properties and
015 * {@link PropertyName} can be used to assign a different local name than the default field name.
016 */
017public class ConfigUtils {
018
019  /**
020   * Builds properties from a configuration bean using no namespace prefix.
021   *
022   * @param config the configuration object to inspect
023   *
024   * @return properties instance derived
025   */
026  public static Properties toProperties(Object config) {
027    return toProperties("", config);
028  }
029
030  /**
031   * Builds properties from a configuration bean using a given namespace prefix which is added to all properties.
032   *
033   * @param config the configuration object to inspect
034   *
035   * @return properties instance derived
036   */
037  public static Properties toProperties(String prefix, Object config) {
038    Properties props = new Properties();
039
040    Field[] fields = config.getClass().getDeclaredFields();
041    for (Field f : fields) {
042      // ignore private fields
043      if (Modifier.isPublic(f.getModifiers()) && f.getAnnotation(IgnoreProperty.class) == null) {
044        try {
045          final String localName =
046            f.getAnnotation(PropertyName.class) == null ? f.getName() : f.getAnnotation(PropertyName.class).value();
047          final String qualName = Strings.isNullOrEmpty(prefix) ? localName : prefix + "." + localName;
048          Object val = f.get(config);
049          if (val == null) {
050            val = "";
051          }
052          // a gbif class? then we need to recursively inspect deeper
053          if (f.getType().getCanonicalName().startsWith("org")) {
054            Properties subProps = toProperties(qualName, val);
055            props.putAll(subProps);
056          } else if (!Collection.class.isAssignableFrom(f.getType())) {
057            props.setProperty(qualName, val.toString());
058          }
059        } catch (IllegalAccessException ignored) {
060          // should never get here cause its accessible
061        }
062      }
063    }
064
065    return props;
066  }
067
068  private ConfigUtils() {
069    throw new UnsupportedOperationException("Can't initialize class");
070  }
071
072}

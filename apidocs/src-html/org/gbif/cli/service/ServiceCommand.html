001package org.gbif.cli.service;
002
003import org.gbif.cli.BaseCommand;
004
005import com.google.common.util.concurrent.Service;
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008
009/**
010 * This provides a convenient way to manage long-running services from a CLI. It is based on the {@link BaseCommand}
011 * and thus inherits all its possibilities to configure a command.
012 * <p/>
013 * You mostly need to implement the {@link #getService()} method to return a Guava {@link Service} class. This will
014 * then be started and stopped on JVM shutdown.
015 */
016public abstract class ServiceCommand extends BaseCommand {
017
018  private static final Logger LOG = LoggerFactory.getLogger(ServiceCommand.class);
019
020  protected ServiceCommand(String name) {
021    super(name);
022  }
023
024  /**
025   * This method will be called after the command line arguments and the configuration file have been processed.
026   * <p/>
027   * The service will be started and on shutdown of the JVM it will also try to execute an orderly shutdown of the
028   * service (using {@link Service#startAsync()} and {@link Service#stopAsync()} respectively).
029   */
030  protected abstract Service getService();
031
032  @Override
033  protected final void doRun() {
034    final Service service = getService();
035
036    Runtime.getRuntime().addShutdownHook(new ShutdownThread(service));
037
038    LOG.info("Service starting ...");
039    try {
040      service.startAsync();
041      service.awaitRunning();
042      LOG.info("Service started");
043    } catch (IllegalStateException e) {
044      LOG.warn("Service failed to start", e);
045    } catch (RuntimeException e) {
046      LOG.warn("Service failed to start", e);
047    }
048  }
049
050  /**
051   * This thread is intended to be passed to {@link Runtime#addShutdownHook(Thread)} and it tries to stop the service
052   * that it's being passed.
053   */
054  @SuppressWarnings("ClassExplicitlyExtendsThread")
055  private class ShutdownThread extends Thread {
056
057    private final Service service;
058
059    private ShutdownThread(Service service) {
060      this.service = service;
061    }
062
063    @Override
064    public void run() {
065      LOG.debug("Shutdown Hook called");
066      try {
067        if (service.isRunning()) {
068          LOG.info("Service stopping ...");
069          service.stopAsync();
070          // TODO: Maybe wait with a timeout here after which we force close?
071          service.awaitTerminated();
072          LOG.info("Service stopped");
073        }
074      } catch (Exception e) {
075        LOG.debug("Caught exception in shutdown hook", e);
076      }
077    }
078  }
079}

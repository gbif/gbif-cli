1   package org.gbif.cli.converter;
2   
3   import java.util.UUID;
4   
5   import com.beust.jcommander.IStringConverter;
6   import com.beust.jcommander.ParameterException;
7   
8   /**
9    * Converts parameters into UUIDs.
10   */
11  public class UuidConverter implements IStringConverter<UUID> {
12    @Override
13    public UUID convert(String value) {
14      try {
15        return UUID.fromString(value);
16      } catch(IllegalArgumentException ex) {
17        throw new ParameterException(value + " is not a valid uuid");
18      }
19    }
20  }

51    /**
52     * Main entry point. This method searches for other commands on the classpath using the {@link ServiceLoader} and
53     * parses the command line to see which command to run.
54     *
55     * @param arguments command line arguments
56     *
57     * @return exit code
58     */
59    @SuppressWarnings("UseOfSystemOutOrSystemErr")
60    public final int run(String... arguments) {
61      configureLogback();
62  
63      // Try to load additional commands
64      ServiceLoader<Command> loader = ServiceLoader.load(Command.class);
65      for (Command command : loader) {
66        addCommand(command);
67      }
68  
69      initialize();
70  
71      if (arguments.length == 0) {
72        printUsage(System.err);
73        return 1;
74      }
75  
76      // Command has to be the very first argument
77      String commandName = arguments[0];
78      if (!commands.containsKey(commandName)) {
79        System.err.println("The command name you supplied is not a valid command.");
80        System.err.println("You supplied: [" + commandName + "]");
81        if (commands.isEmpty()) {
82          System.err.println("There are no valid commands. This is most likely a programming error.");
83          System.err.println("Hint: Maybe you forgot to annotate your commands with '@MetaInfServices(Command.class)'?");
84        } else {
85          System.err.println("The valid commands are:");
86        }
87        for (String availableCommand : commands.keySet()) {
88          System.err.println(" - " + availableCommand);
89        }
90        return 1;
91      }
92  
93      // Copy the remaining arguments (minus command name)
94      String[] commandArguments = Arrays.copyOfRange(arguments, 1, arguments.length);
95  
96      Command command = commands.get(commandName);
97      try {
98        command.run(commandArguments);
99      } catch (Exception t) {
100       System.err.println("Command threw exception");
101       t.printStackTrace();
102       return 1;
103     }
104     return 0;
105   }

107   /**
108    * This checks if the user has provided a logback configuration file using the usual logback loading mechanism. If he
109    * does not logback will already have run {@link ch.qos.logback.classic.BasicConfigurator#configureDefaultContext();}
110    * which means we're logging to the console at DEBUG level. To prevent that we try the same mechanism that logback
111    * uses to find a configuration file. If it can't find this file then it resets logback so that it doesn't print
112    * anything.
113    * <p/>
114    * The drawback here is that it doesn't take into account the configurability of logback files using the BaseCommand.
115    */
116   private void configureLogback() {
117     LoggerContext context = (LoggerContext) LoggerFactory.getILoggerFactory();
118     ContextInitializer ci = new ContextInitializer(context);
119     URL loggingConfigUrl = ci.findURLOfDefaultConfigurationFile(false);
120 
121     if (loggingConfigUrl == null) {
122       context.reset();
123     }
124   }

126   private void printUsage(OutputStream os) {
127     PrintWriter pw = new PrintWriter(os);
128     try {
129       pw.println("Usage");
130 
131       for (Map.Entry<String, Command> commandEntry : commands.entrySet()) {
132         pw.println();
133         pw.append("Command: ").println(commandEntry.getKey());
134         pw.println("---------------------------------------");
135 
136         Optional<String> usage = commandEntry.getValue().getUsage();
137         pw.println(usage.or("This command does not have any options"));
138       }
139     } catch (Exception e) {
140       System.err.println("Exception during generation of usage instructions");
141       e.printStackTrace();
142     } finally {
143       pw.flush();
144       Closeables.closeQuietly(pw);
145     }
146   }

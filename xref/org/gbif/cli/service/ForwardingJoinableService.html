9   /**
10   * This class wraps another service (using Guava's {@link ForwardingService} to provide a way to {@link #join()} a
11   * service and wait for its termination.
12   */
13  class ForwardingJoinableService extends ForwardingService {
14  
15    private static final Logger LOG = LoggerFactory.getLogger(ForwardingJoinableService.class);
16  
17    private final Object joinLock = new Object();
18  
19    private final Service delegate;
20  
21    protected ForwardingJoinableService(Service delegate) {
22      this.delegate = delegate;
23  
24      addListener(new Listener() {
25        @Override
26        public void starting() {
27          LOG.debug("New state: [STARTING]");
28        }
29  
30        @Override
31        public void running() {
32          LOG.debug("New state: [RUNNING]");
33        }
34  
35        @Override
36        public void stopping(State from) {
37          LOG.debug("New state: [STOPPING] from [{}]", from);
38        }
39  
40        @Override
41        public void terminated(State from) {
42          LOG.debug("New state: [TERMINATED] from [{}]", from);
43  
44          synchronized (joinLock) {
45            joinLock.notifyAll();
46          }
47        }
48  
49        @Override
50        public void failed(State from, Throwable failure) {
51          LOG.debug("New state: [FAILED] from [{}]", from, failure);
52  
53          synchronized (joinLock) {
54            joinLock.notifyAll();
55          }
56        }
57      }, MoreExecutors.sameThreadExecutor());
58  
59    }
60  
61    @Override
62    protected Service delegate() {
63      return delegate;
64    }
65  
66    /**
67     * Waits for this service to exit RUNNING state.
68     */
69    public void join() throws InterruptedException {
70      LOG.debug("Thread joined service");
71      synchronized (joinLock) {
72        while (isRunning()) {
73          LOG.debug("Calling wait");
74          joinLock.wait();
75        }
76      }
77      LOG.debug("Thread returning from join");
78    }

80  }

003import java.io.File;
004import java.io.IOException;
005import java.net.MalformedURLException;
006import java.net.URL;
007import java.util.List;
008import java.util.Set;
009import javax.validation.ConstraintViolation;
010import javax.validation.Validation;
011import javax.validation.Validator;
012
013import ch.qos.logback.classic.Level;
014import ch.qos.logback.classic.LoggerContext;
015import ch.qos.logback.classic.joran.JoranConfigurator;
016import ch.qos.logback.core.joran.spi.JoranException;
017import ch.qos.logback.core.util.Loader;
018import ch.qos.logback.core.util.StatusPrinter;
019import com.beust.jcommander.JCommander;
020import com.beust.jcommander.ParameterException;
021import com.fasterxml.jackson.databind.DeserializationFeature;
022import com.fasterxml.jackson.databind.ObjectMapper;
023import com.fasterxml.jackson.databind.ObjectReader;
024import com.fasterxml.jackson.databind.jsonschema.JsonSchema;
025import com.fasterxml.jackson.dataformat.yaml.YAMLFactory;
026import com.fasterxml.jackson.dataformat.yaml.snakeyaml.parser.ParserException;
027import com.google.common.base.Optional;
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030import org.slf4j.bridge.SLF4JBridgeHandler;

032/**
033 * This base command can be used for all Commands that want to have their parameters validated and who want to allow
034 * specifying configuration data in a JSON or YAML file.
035 * <p/>
036 * JSR 303 annotations can be used for validation as well as the built-in methods from JCommander.
037 */
038@SuppressWarnings("UseOfSystemOutOrSystemErr")
039public abstract class BaseCommand extends Command {
040
041  private static final Logger LOG = LoggerFactory.getLogger(BaseCommand.class);
042
043  private static final ObjectMapper MAPPER = new ObjectMapper(new YAMLFactory());
044
045  private static final Validator VALIDATOR = Validation.buildDefaultValidatorFactory().getValidator();
046
047
048  static {
049    MAPPER.disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES);
050  }
051
052  private final String name;
053
054  protected BaseCommand(String name) {
055    super(name);
056    this.name = name;
057  }
058
059  /**
060   * The object returned by this method will be populated from the JSON or YAML files specified using the {@code --conf}
061   * command line parameter. The files given in this way will be applied in the order they appeared on the command line
062   * so later files might overwrite earlier ones.
063   * <p/>
064   * This can be the same object as the one returned from {@link #getParameterObject()} in which case command line
065   * parameters override options from the files.
066   */
067  protected abstract Object getConfigurationObject();
068
069  /**
070   * This is the object which has JCommander annotations and will be populated from the command line arguments.
071   * <p/>
072   * This can be the same object as the one returned from {@link #getConfigurationObject()}} in which case command line
073   * parameters override options from the files. The default implementation defers to {@link #getConfigurationObject()}.
074   */
075  protected Object getParameterObject() {
076    return getConfigurationObject();
077  }
078
079  /**
080   * This method will be called when the generic parameters have been handled and configuration and parameter objects
081   * have been populated.
082   */
083  protected abstract void doRun();
084
085  @Override
086  public Optional<String> getUsage() {
087    JCommander jCommander = new JCommander();
088    jCommander.setProgramName(name);
089    jCommander.addObject(new GenericParameters());
090    jCommander.addObject(getParameterObject());
091
092    StringBuilder sb = new StringBuilder();
093    jCommander.usage(sb);
094
095    String sep = System.getProperty("line.separator");
096    try {
097      JsonSchema jsonSchema = MAPPER.generateJsonSchema(getConfigurationObject().getClass());
098      sb.append(sep);
099      sb.append("The configuration file has the following JSON schema but accepts either YAML or JSON:");
100      sb.append(sep);
101      sb.append(jsonSchema);
102    } catch (Exception e) {
103      LOG.debug("Error generating JSON schema, this is most likely a programming error", e);
104    }
105
106    return Optional.of(sb.toString());
107  }
108
109  @SuppressWarnings("UseOfSystemOutOrSystemErr")
110  @Override
111  public final void run(String... arguments) {
112    // First we process the generic parameters which might cause us to bail out early if there is an error or if the
113    // user requested us to print usage instructions
114    GenericParameters genericParameters = new GenericParameters();
115    JCommander jCommander = new JCommander(genericParameters);
116    jCommander.setAcceptUnknownOptions(true);
117    processCommandLineParameters(jCommander, arguments);
118    if (genericParameters.help) {
119      System.out.println(getUsage().get());
120      return;
121    }
122
123    processConfigurationFiles(genericParameters.configurationFiles);
124
125    List<String> remainingArgs = jCommander.getUnknownOptions();
126    jCommander = new JCommander(getParameterObject());
127    processCommandLineParameters(jCommander, remainingArgs.toArray(new String[remainingArgs.size()]));
128
129    configureLogging(genericParameters);
130    validateObjects();
131
132    doRun();
133  }
134
135  private void processCommandLineParameters(JCommander jCommander, String... arguments) {
136    try {
137      jCommander.parse(arguments);
138    } catch (ParameterException e) {
139      System.err.println("Error parsing command line options:");
140      System.err.println(e.getMessage());
141      throw new CommandException("Error during command line parsing", e);
142    }
143  }
144
145  /**
146   * We take every passed in configuration file, test if it exists and try to write it to the Object provided by {@link
147   * #getConfigurationObject()}. This method prints errors it encounters.
148   *
149   * @param configurationFiles to process
150   */
151  private void processConfigurationFiles(Iterable<String> configurationFiles) {
152    for (String fileName : configurationFiles) {
153      File file = new File(fileName);
154      if (!file.exists()) {
155        System.err.println("Error reading configuration file [" + fileName + "] because it does not exist");
156        throw new CommandException("Error reading configuration file [" + fileName + "] because it does not exist");
157      }
158      ObjectReader reader = MAPPER.readerForUpdating(getConfigurationObject());
159      try {
160        reader.readValue(file);
161      } catch (IOException e) {
162        System.err.println("Error reading configuration file [" + fileName + "]");
163        throw new CommandException("Error reading configuration file [" + fileName + "]", e);
164      } catch (ParserException e) {
165        System.err.println("Error reading configuration file [" + fileName + "]");
166        throw new CommandException("Error reading configuration file [" + fileName + "]", e);
167      }
168    }
169  }
170
171  /**
172   * Reconfigures Logback.
173   * <p/>
174   * Usually logback loads a <em>logback.xml</em> file but we change that. We allow a command line parameter to
175   * override the config file to use. If that isn't specified or doesn't exist we try to load a file called
176   * <em>logback-$commandName.xml</em> and if that too doesn't exist we are using the default.
177   * <p/>
178   * Additionally we set the ROOT log level to the one that was provided, if any, or to DEBUG if requested otherwise we
179   * leave it at the default.
180   */
181  private void configureLogging(GenericParameters parameters) {
182    // Set up JUL logging to go through SLF4J
183    SLF4JBridgeHandler.removeHandlersForRootLogger();
184    SLF4JBridgeHandler.install();
185
186    URL loggingConfigUrl = null;
187
188    // Try to load a logback configuration file from the file system if given on the command line
189    if (parameters.logbackConfig != null && !parameters.logbackConfig.isEmpty()) {
190      File configFile = new File(parameters.logbackConfig);
191      if (!configFile.exists() || !configFile.isFile()) {
192        throw new CommandException("Logback configuration file does not exist");
193      }
194      try {
195        loggingConfigUrl = configFile.toURI().toURL();
196      } catch (MalformedURLException e) {
197        // This should not happen because we already checked for existence
198        LOG.debug("Failed to create URL for logback configuration file");
199        throw new CommandException(e);
200      }
201    }
202
203    // Try to automatically load a logback configuration file named after the command
204    if (loggingConfigUrl == null) {
205      loggingConfigUrl = Loader.getResource("logback-" + getName() + ".xml", Loader.getClassLoaderOfObject(this));
206    }
207
208    // If we got a configuration file reconfigure logback now
209    if (loggingConfigUrl != null) {
210      LoggerContext context = (LoggerContext) LoggerFactory.getILoggerFactory();
211      context.reset();
212      JoranConfigurator joran = new JoranConfigurator();
213      joran.setContext(context);
214      try {
215        joran.doConfigure(loggingConfigUrl);
216      } catch (JoranException e) {
217        StatusPrinter.printInCaseOfErrorsOrWarnings(context);
218      }
219    }
220
221    // Change log levels if needed
222    if (parameters.verbose || (parameters.logLevel != null && !parameters.logLevel.isEmpty())) {
223      ch.qos.logback.classic.Logger root =
224        (ch.qos.logback.classic.Logger) LoggerFactory.getLogger(Logger.ROOT_LOGGER_NAME);
225      root.setLevel(Level.toLevel(parameters.logLevel));
226
227      if (parameters.verbose) {
228        root.setLevel(Level.DEBUG);
229      }
230    }
231  }
232
233  /**
234   * This validates the parameter and configuration object and prints violations if there are some. If the objects are
235   * exactly the same it will only be validated once.
236   */
237  @SuppressWarnings("ObjectEquality")
238  private void validateObjects() {
239    validateObject(VALIDATOR, getConfigurationObject());
240    if (getParameterObject() != getConfigurationObject()) {
241      validateObject(VALIDATOR, getParameterObject());
242    }
243  }
244
245  /**
246   * Validates a single object, printing any violations it encounters.
247   *
248   * @param validator to use
249   * @param object    to validate
250   */
251  private void validateObject(Validator validator, Object object) {
252    Set<ConstraintViolation<Object>> violations = validator.validate(object);
253    if (!violations.isEmpty()) {
254      System.err.println("Failed validation of command line parameters:");
255      for (ConstraintViolation<Object> violation : violations) {
256        System.err.println("  " + violation.getPropertyPath() + ": " + violation.getMessage());
257      }
258      throw new CommandException("Failed validation");
259    }
260  }

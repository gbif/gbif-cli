001package org.gbif.cli;
002
003import com.google.common.base.Optional;
004
005import static com.google.common.base.Preconditions.checkArgument;
006import static com.google.common.base.Preconditions.checkNotNull;
007
008/**
009 * Base class for commands which needs to be subclassed.
010 */
011public abstract class Command {
012
013  private final String name;
014
015  protected Command(String name) {
016    this.name = checkNotNull(name, "name can't be null");
017    checkArgument(!name.isEmpty(), "name can't be empty");
018  }
019
020  public final String getName() {
021    return name;
022  }
023
024  /**
025   * Optionally returns a string describing the usage of this command.
026   *
027   * @return {@link Optional#absent()} if this command doesn't take any further parameters otherwise a string
028   *         describing
029   *         the parameters
030   */
031  public abstract Optional<String> getUsage();
032
033  /**
034   * This is the main entry point into this command.
035   *
036   * @param arguments all arguments that are relevant for this command. This does not include the command name or any
037   *                  other arguments that the Application itself might have parsed
038   */
039  public abstract void run(String... arguments);
040
041}
